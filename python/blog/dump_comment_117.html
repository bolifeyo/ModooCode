    #include <stdio.h>

    char change(char* a, char* b);


    int main()
    {
    FILE* p=fopen("a.txt","r");
    FILE* fp=fopen("b.txt","w");

    char str[30];
    char str2[30];

    fgets(str,30,p);

    printf("%s\n",str);

    change(str2,str);
    printf("%s",str2);

    fputs(str2,fp);


    fclose(p);
    fclose(fp);

    return 0;
    }

    char change(char* a, char* b)
    {
    int count=0;

    while(*b)
    {
    b++;
    count++;
    }

    b--;

    while(count>=0)
    {
    *a=*b;
    a++;
    b--;
    count--;
    }

    *a='\0';


    return 0;
    }
