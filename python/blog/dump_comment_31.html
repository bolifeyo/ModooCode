    strtok함수 소스를 짜보았는데요.. 이상한 에러가 나요..ㅠ
    한번 봐주시면 감사하겠습니다..ㅠㅠ

    #include <stdio.h>
    #include <string.h>

    char* strtok(char* str, char* delim);

    int main(void)
    {
    char str[] = "010-1111-3333";
    char* delim = "-";
    char* token;

    token = strtok(str, delim);

    while(token!=NULL)
    {
    puts(token);
    token=strtok(NULL,delim);
    }
    return 0;
    }

    char* strtok(char* str, char* delim)
    {
    char* cntstr=str;
    static char* nextstr=NULL;
    int len=0;
    int i=0;
    if(str == NULL) //첫번째 이후의 토큰을 찾을 때
    cntstr = nextstr; //다음 토큰을 현재 토큰에 저장

    len = strlen(cntstr);
    if(len == 0) //토큰의 길이가 0, 즉 문자열의 끝을 의미
    {
    return NULL; //NULL반환
    }

    for(i=0; i<len+1; i++) //현재 cntstr부터 하나씩 차례로
    { //구분기호나 NULL이 나올 때까지 검사
    if(*(cntstr+i) == delim[0])
    {
    nextstr = cntstr+i+1; //구분기호가 나올 경우 다음 토큰의 위치를 정해주고
    *(cntstr+i) = NULL; //구분기호는 NULL로 변경
    break; //FOR문 이탈
    }
    else if(*(cntstr+i) == NULL)
    {
    nextstr = NULL; //문자열의 끝이므로 다음 토큰은 문자를 가리키지 않고 NULL을 저장
    break; //FOR문 이탈
    }
    }
    return cntstr;
    }
