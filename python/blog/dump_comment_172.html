    day += inc;
    while(1){
    if(year % 4 == 0) days[2] = 29;
    else days[2] = 28;

    if(day > days[month]){
    day -= days[month];
    month++;
    if(month == 13){
    year++;
    month = 1;
    }
    }
    else break;
    }

    year += month / 12;

    }
