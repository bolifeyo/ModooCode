  6. 질문이요!

    class A
    {
    int x;

    public:
    A(int c) : x(c) {}
    A(const A& a) { x = a.x; cout << "복사생성" << endl;}
    };

    class B
    {
    A a;

    public:
    B(int c) : a(c) {}
    B(const B& b) : a(b.a) { }
    A get_A() { return a;}
    };

    int main()
    {
    B b(10);
    A a1 = b.get_A();

    }
    B b(10); 에서 초기화리스트 방식에 따라 기본 생성자가 아닌(선언도 안되어 있지만) 인자 1나를 갖는 a의 생성자가 먼저 실행되고 그다음에 b생성자가 실행이 됩니다.

    A a1 = b.get_A(); 에서는 b.get_A()이므로 reference가 아니기 때문에 임시로 한번 복사가 일어납니다. 그 다음에 A a1에 의해 복사 연산자가 한번 더 실행됩니다. 답은 2번. 하지만 copy elision에 의해 1번만 일어나네요.
