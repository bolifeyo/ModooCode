    #include <iostream>
    #include <math.h>
    using namespace std;

    class Point {
    public:
    int x, y;
    Point(int pos_x, int pos_y) {
    x = pos_x;
    y = pos_y;
    }
    Point() {}
    };

    class Geometry {
    int now = 0;
    // 현재 공간에 대한 점들의 정보를 담고 있는 배열
    // 물론 배열의 크기는 생성자에서 초기화 하고 충분히
    // 크게 잡도록 합시다 (점 100 개 정도?)
    Point *pointArray; // Point pointArray[];
    public:
    Geometry(Point *pointList, int length) {
    pointArray = pointList;
    now = length;
    }
    Geometry() { // 필요한가?
    pointArray = new Point[100]; // pointArray = (Point *)malloc(sizeof(Point) * 100);
    }

    void addPoint(Point &point);

    // 모든 점들 간의 거리를 출력하는 함수 입니다.
    void printDistance();

    // 모든 점들을 잇는 직선들 간의 교점의 수를 출력해주는 함수 입니다.
    // 참고적으로 임의의 두 점을 잇는 직선의 방정식을 f(x,y) = ax+by+c = 0 <<기울기가 무한대인 경우 정의불가
    // 이라고 할 때 임의의 다른 두 점 (x1, y1) 과 (x2, y2) 가 f(x,y)=0 을 기준으로
    // 서로 다른 부분에 있을 조건은 f(x1, y1) * f(x2, y2) <= 0 이면 됩니다.
    void printNumMeets();
    };

    void Geometry::addPoint(Point &point) {
    pointArray[now] = point;
    now++;
    }

    void Geometry::printDistance() {
    for (int i = 0; i < now; i++) {
    for (int j = i + 1; j < now; j++) {
    cout << i + 1 << "번과 " << j + 1 << "번의 거리 : "
    << sqrt((pointArray[i].x - pointArray[j].x)*(pointArray[i].x - pointArray[j].x)
    + (pointArray[i].y - pointArray[j].y)*(pointArray[i].y - pointArray[j].y)) << endl;
    }
    }
    }

    void Geometry::printNumMeets() {
    int meets = 0;

    for (int i = 0; i < now; i++) {
    for (int j = i + 1; j < now; j++) {
    for (int k = 0; k < now; k++) {
    for (int l = k + 1; l < now; l++) {
    if (k != i && k != j && l != i && l != j) {
    if (
    (
    (pointArray[i].x - pointArray[j].x)*(pointArray[k].y - pointArray[i].y)
    - (pointArray[i].y - pointArray[j].y)*(pointArray[k].x - pointArray[i].x)
    )
    *
    (
    (pointArray[i].x - pointArray[j].x)*(pointArray[l].y - pointArray[i].y)
    - (pointArray[i].y - pointArray[j].y)*(pointArray[l].x - pointArray[i].x)
    )
    <= 0
    )
    meets++;
    }
    }
    }
    }
    }
    cout << "교점의 개수 : " << meets / 2 << endl;
    }

    }
    else {
    if (num >= 100) {
    cout << "점이 너무 많아" << endl;
    }
    break;
    }
    }
    geo.printDistance();
    geo.printNumMeets();

    cin >> choice;
    if (choice != 1) {
    break;
    }
    }
    delete[]points;

    }
