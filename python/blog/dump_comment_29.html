    printf("같다\n");
    }else{
    printf("다르다\n");
    }
    }

    int IsEqual(char *str_a, char * str_b){
    int i=0;
    while(str_a[i] != NULL || str_b[i] != NULL){
    if(str_a[i] != str_b[i]){
    return 0;

    i++;
    }
    return 1;
    }
