    #include<stdio.h>
    #include<math.h>
    #pragma warning(disable:4996)

    int main()
    {
    int i;
    double num;
    int arr[1000];
    printf("몇까지? \n");
    scanf("%f", &num);

    for (i = 2; i <= num; i++)
    {

    arr[i] = i;

    }



    for (i = 2; i <= sqrt(num); i++)
    {

    if (arr[i] == 0) // 이미 체크된 수의 배수는 확인하지 않는다

    continue;

    for (int j = i + i; j <= num; j += i)
    { // i를 제외한 i의 배수들은 0으로 체크

    arr[j] = 0;

    }

    }


    // print

    for (i = 2; i <= num; i++)
    {

    if (arr[i] != 0)

    printf("소수는 %d",arr[i]);

    }

    return 0;
    }
